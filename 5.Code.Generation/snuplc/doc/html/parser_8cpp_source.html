198 CAstStatement* CParser::statSequence(CAstScope *s)
199 {
200  //
201  // statSequence ::= [ statement { ";" statement } ].
202  // statement ::= assignment | subroutineCall | ifStatement | whileStatement | returnStatement
203  // FIRST = FIRST(<statement>) U {e}
204  // FOLLOW = {tEnd, tElse}
205  //
206 
207  CAstStatement *head = NULL;
208  CSymtab *table = s->GetSymbolTable(); // get symbol table
209 
210  EToken tt = _scanner->Peek().GetType(); // peek next token
211  if (!(tt == tEnd || tt == tElse)) { // while current token isn't included in FOLLOW
212  CAstStatement *tail = NULL;
213 
214  do {
215  CToken t;
216  EToken tt = _scanner->Peek().GetType();
217  CAstStatement *st = NULL;
218 
219  switch (tt) {
220  /*
221  * statement = assignment | subroutineCall | ifStatement | whileStatement | returnStatement
222  * FIRST(<statement>) = {tIdent, tIdent, tIf, tWhile, tReturn}
223  */
224 
225  case tIdent:
226  // if tIdent's value type is procedure it is subroutinecall
227  if (table->FindSymbol(_scanner->Peek().GetValue()) == NULL)
228  SetError(_scanner->Peek(), "No such symbol");
229  else if (table->FindSymbol(_scanner->Peek().GetValue())->GetSymbolType() == stProcedure) {
230  st = subroutinecall(s);
231  }
232  // else it is assignment
233  else {
234  st = assignment(s);
235  }
236 
237  break;
238 
239  // statement ::= ifStatement
240  case tIf:
241  st = ifstatement(s);
242  break;
243 
244  // statement ::= whileStatement
245  case tWhile:
246  st = whilestatement(s);
247  break;
248 
249  // statement ::= returnStatement
250  case tReturn:
251  st = returnstatement(s);
252  break;
253 
254  default:
255  SetError(_scanner->Peek(), "statement expected.");
256  break;
257  }
258 
259  assert(st != NULL); // for error handling
260 
261  if (head == NULL) // if it is first time
262  head = st; // make new statment
263  else
264  tail->SetNext(st); // else append to next
265 
266  tail = st;
267 
268  tt = _scanner->Peek().GetType();
269  if (tt == tEnd || tt == tElse) break;
270 
271  Consume(tSemicolon);
272  } while (!_abort);
273  }
274 
275  return head;
276 }

294 {
295  /*
296  * expression ::= simpleexpr [ relOp simpleexpression ].
297  */
298 
299  CToken t;
300  EOperation relop;
301  CAstExpression *left = NULL, *right = NULL;
302 
303  left = simpleexpr(s);
304  if (_scanner->Peek().GetType() == tRelOp) { // if it has [ relOp simpleexpression ]
305  Consume(tRelOp, &t); // Consume relOp

866 CAstProcedure* CParser::subroutinedecl(CAstScope *s)
867 {
868  //
869  // subroutinedecl = (proceduredecl | functiondecl)
870  // subroutinebody ident ";".
871  // proceduredecl = "procedure" ident [ formalparam ] ";"
872  // functiondecl = "function" ident [ formalparam ] ":" type ";"
873  // formalparam = "(" [ vardeclsequence ] ")".
874  // vardeclsequence = vardecl { ";" vardecl }.
875  // vardecl = ident { "," ident } ":" type
876  //
877 
878  // Pointers for assignments
879  CToken t, tname, tendname, tval;
880  CSymtab *table = s->GetSymbolTable();
881  CAstProcedure *n = NULL;
882  CAstType *typ = NULL;
883  CSymProc *sym = NULL;
884  vector<CSymParam *> v;
885  vector<CToken> vec;
886  int index=0, size;
887 
888  if (_scanner->Peek().GetType() == tProcedure) { // if it is proceduredecl
889  Consume(tProcedure, &t);
890  }
891  else if (_scanner->Peek().GetType() == tFunction) { // if it is functiondecl
892  Consume(tFunction, &t);
893  }
894  else { // else raise error
895  SetError(_scanner->Peek(), "\"procedure\" or \"function\" expected.");
896  }
897 
898  Consume(tIdent, &tname); // consumes ident
899 
900  if (table->FindSymbol(tname.GetValue()) != NULL){
901  SetError(tname, "duplicate procedure/function declaration");
902  }
903  // formalparam
904  if (_scanner->Peek().GetType() == tLBrak) {
905  // Consumes keywords
906  Consume(tLBrak);
907 
908  // Untill we find right bracket
909  if (_scanner->Peek().GetType() != tRBrak) {
910  Consume(tIdent, &tval); // Consumes keywords
911  vec.push_back(tval); // Put in vector
912  while (_scanner->Peek().GetType() == tComma) { // if it has other ident, consumes it
913  Consume(tComma);
914  Consume(tIdent, &tval);
915  for (int i=0;i<vec.size(); i++){
916  if (vec[i].GetValue().compare(tval.GetValue())==0){
917  SetError(tval, "duplicated variables");
918  }
919  }
920  vec.push_back(tval);
921  }
922  Consume(tColon); // Consuems colon
923  typ = type(s, false, true);
924  size = vec.size();
925 
926  for (int i=0; i<size; i++) { // Iterate vector and append its elements to another vector
927  v.push_back(new CSymParam(index, vec[i].GetValue(), typ->GetType()));
928  index++;
929  }
930 
931  while (_scanner->Peek().GetType() == tSemicolon) {
932  vec.clear(); // Repeatedly do the same job while nextType is ";"
933  Consume(tSemicolon); // Consumes keywords
934  Consume(tIdent, &tval);
935  vec.push_back(tval);
936 
937  for (int i=0;i<v.size(); i++){
938  if (tval.GetValue().compare(v[i]->GetName()) == 0){
939  SetError(tval, "duplicated variables");
940  }
941  }
942  while (_scanner->Peek().GetType() == tComma) {
943  Consume(tComma); // Consuems keywords
944  Consume(tIdent, &tval);
945  vec.push_back(tval);
946 
947  for (int i=0;i<v.size();i++){
948  if (tval.GetValue().compare(v[i]->GetName()) == 0){
949  SetError(tval, "duplicated variables");
950  }
951  }
952  }
953  Consume(tColon); // Consumes keywords
954  typ = type(s, false, true);
955  size = vec.size();
956 
957  for (int i=0; i<size; i++) { // Iterate vector and append its elements to another vector
958  v.push_back(new CSymParam(index, vec[i].GetValue(), typ->GetType()));
959  index++;
960  }
961  }
962  }
963  Consume(tRBrak); // Finally consumes right bracket
964  }
965 
966  if (t.GetType() == tFunction) { // if it is functiondecl
967  Consume(tColon); // consuems colon
968  typ = type(s, true); // it should check [] is empty
969  if (!typ->GetType()->IsScalar())
970  SetError(typ->GetToken(), "return type should be scalar");
971  sym = new CSymProc(tname.GetValue(), typ->GetType());
972  }
973  else { // if it is proceduredecl
974  sym = new CSymProc(tname.GetValue(), CTypeManager::Get()->GetNull());
975  }
976  Consume(tSemicolon); // consuems semicolon
977 
978  size = v.size();
979  for (int i=0;i<size;i++) { // add all parameters to the symbol tree
980  sym->AddParam(v[i]);
981  }
982 
983  n = new CAstProcedure(t, tname.GetValue(), s, sym);
984  table->AddSymbol(sym);
985  table = n->GetSymbolTable(); // add all parameters to the symbol tree
986  for (int i=0;i<size;i++) {
987  table->AddSymbol(v[i]);
988  }
989 
990  // subroutinebody
991 
992  subroutinebody(n);
993  Consume(tIdent, &tendname);
994  Consume(tSemicolon);
995 
996  // different ident error handling
997  if (tendname.GetValue().compare(tname.GetValue()) != 0) {
998  SetError(tendname, "expected \'" + tname.GetValue() + "\', end with \'" + tendname.GetValue() + "\'");
999  }
1000 
1001  return n;
1002 }
